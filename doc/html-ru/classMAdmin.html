Подробное описание

Модуль управления админом контракта

Админ или владелец может заменить админа на другого Пример использование модуля без других модулей доступа

#Include "module/MAdmin.ligo"
type t_storage record [
admin: MAdmin.t_admin;
...
];
type t_entrypoint is
| ChangeAdmin of MAdmin.t_admin
...
function main(const entrypoint: t_entrypoint; var s: t_storage): t_return is
case entrypoint of
| ChangeAdmin(params) -> (cNO_OPERATIONS, { s.admin:= MAdmin.accessChange(params, s.admin); } with s)
...
